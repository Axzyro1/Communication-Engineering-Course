#include <stdio.h>
int main()
{
    int n,i,j;
    while(scanf("%d",&n)!=EOF)
    {
    	for(j=1;j<=n;j++)
		{
		scanf("%d",&i);
		if(i<=100)
		printf("%d\n",85*i);
		if(i>100&&i<=500)
		printf("%d\n",70*i);
		if(i>500&&i<=1000)
		printf("%d\n",60*i);
		if(i>1000&&i<=5000)
		printf("%d\n",50*i);
		if(i>5000)
		printf("%d\n",30*i);
		}
  }
}
