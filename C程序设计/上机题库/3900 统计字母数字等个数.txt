3900 Problem I 零起点学算法101――统计字母数字等个数 
#include<stdio.h>
#include <string.h>
int main()
{
 char str[1000];
 int a,b,c,d,i,m;
 while(gets(str)!=NULL)
 {
 	a=b=c=d=0;
    m=strlen(str);
 	for(i=0;i<m;i++)
	{
	
	if(str[i]>='a'&& str[i]<='z'||str[i]>='A'&&str[i]<='Z')
    a++;
    else
       if(str[i]==' ')
        b++;
      else
        if(str[i]>='0'&&str[i]<='9')
        c++;
        else
        d++;
    }
 printf("%d ",a);
 printf("%d ",c);
 printf("%d ",b);
 printf("%d\n",d);
}
}
