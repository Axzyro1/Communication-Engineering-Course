#include<stdio.h>
struct student
{
	char number[10];
	char name[10];
	float score[5],ave[5];
};
main()
{
struct student a;
	int i,j;
	for(j=0;j<3;j++)
	{
	  scanf("%d%s",&a.number,a.name);
	  a.ave[j]=0;
	  for(i=0;i<2;i++)
          {
	    scanf("%f",&a.score[i]);
	    a.ave[j]+=a.score[i]/2;
	  }
  	}
  	for(j=0;j<3;j++)
	{
	printf("ave[%d]=%.1f",j,a.ave[j]);
	printf("\n");
	}
}
