3895 Problem D 零起点学算法94――输出矩阵 
#include<stdio.h>
#define N 20
int main()
{
  int m,n,i,j,a[N][N];
  while(scanf("%d%d",&m,&n)!=EOF)
  {
    for(i=0;i<m;i++)
     {
        for(j=0;j<n;j++)
         {
           a[i][j]=1+n*i+j;
         }
     }
      for(i=0;i<m;i++)
     {
          for(j=0;j<n;j++)
         {
           printf("%d ",a[i][j]);
         }
           printf("\n");
     }
 }
}
