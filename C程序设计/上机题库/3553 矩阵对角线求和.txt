#include <stdio.h>
#define N 100
int main()
{
int n,i,j,a[N][N],c=0,b=0;
scanf("%d",&n);
for(i=0;i<n;i++)
     {
      for(j=0;j<n;j++)
       {
       scanf("%d",&a[i][j]);
       }
     }
for(i=0;i<n;i++)
     {
      c+=a[i][i];	
      b+=a[i][n-i-1];
     }
     printf("%d %d\n",c,b);
}
