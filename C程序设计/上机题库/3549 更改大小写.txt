3549 更改大小写
#include <stdio.h>
#include <string.h>
int main()
{
    char c[1000];
    int len, i;
    gets(c); 
    {
    len = strlen(c);
    for (i=0; i<len; i++)
    {
    	if(c[i]>='a'&&c[i]<='z')
        {printf("%c", c[i]-32);}
        else
        {
        printf("%c",c[i]);	
        }
    }
     printf("\n");
    }
}
