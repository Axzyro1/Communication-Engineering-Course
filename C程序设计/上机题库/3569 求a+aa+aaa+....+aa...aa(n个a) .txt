#include<stdio.h>
main()
{
    int i,s,a,n;
    int ab(int a,int n);
    while(scanf("%d%d",&a,&n)!=EOF)
    {
	s=0;
    for(i=1;i<=n;i++)
		{
        s=s+ab(a,i);
        }
    printf("%d\n",s);
    }
}
 int ab(int a,int n)
{
	int s,x,i,j;
	x=n;
	j=0;
	for(i=1;i<=x;i++){
		j=j*10+a;
	}
	return j;	
}
