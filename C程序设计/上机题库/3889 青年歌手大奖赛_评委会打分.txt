3889 Problem H 零起点学算法88――青年歌手大奖赛_评委会打分 
#include<stdio.h>
#define N 100
int main()
{
	int i=0,n,min,max,a[N];
	float s;
	while(scanf("%d",&n)!=EOF)
	{
		s=max=0;
		for(i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
			s+=a[i];
		}	
		min=a[0];
		for(i=0;i<n;i++)
			 {
			 if (max<a[i])
             max=a[i];
             if (min>a[i])
             min=a[i];
			 }
		s=s-min-max;
		printf("%.2f\n",s/(n-2));
	}
}
