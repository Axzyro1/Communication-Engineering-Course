3888 Problem G 零起点学算法87――打印所有低于平均分的分数 
#include<stdio.h>
#define N 100
int main()
{
	int i=0,j,n,a[N];
	float s,ave;
	while(scanf("%d",&n)!=EOF)
	{
	s=0;j=0;
	for(i=0;i<n;i++)
	  {
	   scanf("%d",&a[i]);
	   s+=a[i];
	  }
    ave=s/n;
	for(i=0;i<n;i++)
	{
		if(a[i]<ave)
		{
		printf("%d ",a[i]);
		j++;  
		}
	}
	printf("\n");
}
}
