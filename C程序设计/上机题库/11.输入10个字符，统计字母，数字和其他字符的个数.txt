#include<stdio.h>
int main(void)
{
   int digit,i,letter,other;
   char ch;
   digit=letter=other=0;
   printf("enter 10 characters:");
for(i=1;i<=10;i++){
  ch=getchar();
  if(ch>='a'&&ch<'z')
  letter++;
  else if(ch>='0'&&ch<='9')
  digit++;
  else
  other++;
}
printf("letter=%d,digit=%d,other=%d\n",letter,digit,other);
  return 0;
}
