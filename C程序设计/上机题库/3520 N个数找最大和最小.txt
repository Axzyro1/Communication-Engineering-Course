#include <stdio.h>
main()
{
     int n,num,max,min,i;
     while(scanf("%d",&n)!=EOF)
     {
	 for (i=n;i>0;i--)
     {
         scanf("%d",&num);
         if (i==n) 
             max=min=num;
         if (max<num)
             max=num;
         if (min>num)
             min=num;
     }
     printf("max=%d, min=%d\n",max,min);
}
}
