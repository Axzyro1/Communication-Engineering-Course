#include<stdio.h>
#define N 100
int main()
{
int i,j,n,x,y;
int a[N];
while(scanf("%d",&n)!=EOF)
{
x=0;
for(i=0;i<n;i++)	
{scanf("%d",&a[i]);}
for(i=0;i<n;i++)
{
if(a[i]>a[x])
{
 x=i;
}
}
y=a[x];
a[x]=a[n-1];
a[n-1]=y;
for(i=0;i<n;i++)	
{printf("%d ",a[i]);}
printf("\n");
}
}
