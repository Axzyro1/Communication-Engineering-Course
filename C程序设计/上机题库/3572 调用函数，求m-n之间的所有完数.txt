#include<stdio.h>
int main()
{
    int factorsum(int x);
	int n,m,i;
    while(scanf("%d %d",&m,&n)!=EOF)
		{	
		  for(i=m;i<n;i++)
		  if(factorsum(i)!=0)
		  {printf("%d ",factorsum(i));}	
		  printf("\n");
		}
}
int factorsum(int x)
{
	int j,sum;
	sum=0;
	for(j=1;j<x;j++)
   	{
    if(x%j==0)
    {sum=sum+j;}
    }
	if(sum==x){return x;}
	else return 0;
}
