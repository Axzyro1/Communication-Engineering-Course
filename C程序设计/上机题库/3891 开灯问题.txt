#include<stdio.h>
#define N 100
int main()
{
	int n,i,j,a[N],m=0;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{a[i]=1;}
	for(i=2;i<=n;i++)
	{
		for(j=i;j<=n;j=j+i)
		{
			if(a[j]==0)
			{a[j]=1;}
			else
			{a[j]=0;}
		}
	}
	for(i=1;i<=n;i++)
	{
		if(a[i]==1)
		{m++;}
	}
	printf("%d\n",m);
}
