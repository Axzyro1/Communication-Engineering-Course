#include <stdio.h>
#include <string.h>
int main()
{
    char c[80];
    int len,i,j;
    while(gets(c)!=NULL)
    {
    	j=0;
		len=strlen(c);
        for (i=0;i<len;i++)
        {
        if(c[i]=='t'&&c[i+1]=='h'&&c[i+2]=='e')
        j++;
       	}
        printf("%d\n",j);
  }
}
