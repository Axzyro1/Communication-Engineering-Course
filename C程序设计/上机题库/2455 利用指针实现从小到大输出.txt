#include <stdio.h>
#define N 20 
void swap(float *p1,float *p2) 
{ 
float t;
t=*p1;
*p1=*p2;
*p2=t;
} 
int main() 
{ 
float a[N];
int i,j,n;  
while(scanf("%d",&n)!=EOF) 
{
for(i=0;i<n;i++)
{
	scanf("%f",&a[i]);
}
for(i=0;i<n-1;i++)
		{
              for(j=0;j<n-i-1;j++)
			{
				if (a[j]>a[j+1])
				{
				swap(&a[j],&a[j+1]);
				}
			}
		} 
for(i=0;i<n;i++) 
printf("%f ",a[i]);
printf("\n"); 
} 
}
