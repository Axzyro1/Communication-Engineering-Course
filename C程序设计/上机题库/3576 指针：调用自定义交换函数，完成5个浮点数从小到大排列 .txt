#include <stdio.h>
#define N 5 
void swap(float *p1,float *p2) 
{ 
float t;
t=*p1;
*p1=*p2;
*p2=t;
} 
int main() 
{ 
float a[N];
int i,j;  
while(scanf("%f%f%f%f%f",&a[0],&a[1],&a[2],&a[3],&a[4])!=EOF) 
{
for(i=0;i<4;i++)
		{
              for(j=0;j<4-i;j++)
			{
				if (a[j]>a[j+1])
				{
				swap(&a[j],&a[j+1]);
				}
			}
		} 
 
printf("%.1f %.1f %.1f %.1f %.1f\n",a[0],a[1],a[2],a[3],a[4]); 
} 
}
