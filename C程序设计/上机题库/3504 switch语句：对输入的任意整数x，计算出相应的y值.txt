#include<stdio.h>
int main()
{
 int x,z;
 float y;
 scanf("%d",&x);
 if(x<0) z=0;        
 if(x>=0&&x<10) z=1;
 if(x>=10&&x<20)  z=2;
 if(x>=20)  z=3;
 switch(z)
 {
   case 0:   y=0;  break;
   case 1:   y=x; break;
   case 2:   y=10; break;
   case 3:   y=30; break;
 }
 printf("y=%.6f\n",y);
}
