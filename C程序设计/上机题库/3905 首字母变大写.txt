#include<stdio.h>
#include <string.h>
main()
{
	char c[100];
	int i,len;
    while(gets(c)!=NULL)
 {
	len=strlen(c);
	if(c[0]>='a'&&c[0]<='z')
	{c[0]=c[0]-32;}
	for(i=1;i<len;i++)
	{
		if(c[i]==' '&&c[i+1]>='a'&&c[i+1]<='z')
		{c[i+1]=c[i+1]-32;}
	}
	for(i=0;i<len;i++)
	{printf("%c",c[i]);}
	printf("\n");
 }
}
