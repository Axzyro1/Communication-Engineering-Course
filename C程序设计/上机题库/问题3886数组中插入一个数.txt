#include<stdio.h>
#define N 100
int main()
{
	int i=0,j=0,m,n,k,a[N],b[N];
	while(scanf("%d",&n)!=EOF)
	{
	for(i=0;i<n;i++)
	  {
	    scanf("%d",&a[i]);
	  }
	scanf("%d%d",&m,&k);
	j=0;
	for(i=0;i<n;i++)
	  {
		if(a[i]<m)
		  {
		  printf("%d ",a[i]);
		  b[j]=a[i];
		  j++;
		  }
		else
		  {
		  break;
		  }
	  }
   printf("%d ",m);
   b[j]=m;
    j++;
    for(;i<n;i++)
      {
	    printf("%d ",a[i]);
	    b[j]=a[i];
	    j++;
      }
      printf("\n");
      printf("%d\n",b[k]);
}
}
