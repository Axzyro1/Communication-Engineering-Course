Description

利用指针实现三个数从大到小输出。 
交换2个数的函数可以利用你在2453上写好的哦。省事多了吧。

Input

多组测试数据，每组一行输入三个数

Output

每组在一行从大到小输出

Sample Input


1 2 3
1 7 5
Sample Output


3 2 1
7 5 1



#include <stdio.h> 
void swap(int *p1,int *p2) 
{ 
int t;
t=*p1;
*p1=*p2;
*p2=t;
} 
int main() 
{ int a,b,c;  
while(scanf("%d%d%d",&a,&b,&c)!=EOF) 
{ 
if(a < b) 
{ 
swap(&a,&b);
}
if(a < c) 
{ 
swap(&a,&c);
} 
if(b < c) 
{ 
swap(&b,&c);
} 
if(a < c) 
{ 
swap(&a,&c);
}  
printf("%d %d %d\n",a,b,c); 
} 
}
