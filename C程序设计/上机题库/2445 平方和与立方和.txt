2445 Problem E 平方和与立方和 
#include"stdio.h" 
int main() 
{ 
 int t,x,y,i,s1,s2; 
 while(scanf("%d%d",&x,&y)!=EOF)
 {
 	{
 	s1=s2=0;
 	if(x>y)
 	 {t=x;x=y;y=t;}
 	else
 	 {x=x;y=y;}
  for(i=x;i<=y;i++)
  if(i%2!=0)
  s1+=(i*i*i);
  else
  s2+=(i*i);
 }
 printf("%d %d\n",s2,s1);
 }
}
