3882 Problem B 零起点学算法81――找出数组中最大元素的位置（下标值） 
#include<stdio.h>
#include<math.h>
int main()
{
int i,j,n,x,y;
int a[10];
while(scanf("%d",&n)!=EOF)
{
x=y=0;
for(i=0;i<n;i++)	
scanf("%d",&a[i]);
for(i=n-1;i>0;i--)
if(a[i]>y)
{
 x=i;
 y=a[i];
}
printf("%d %d",y,x);
printf("\n"); 
}
}
