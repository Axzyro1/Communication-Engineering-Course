3511 Problem G 统计各种字符个数 
#include<stdio.h>
int main()
{
 char m;
 int a=0,b=0,c=0,d=0;
 while((m=getchar())!='\n')
 {
    if(m>='a'&&m<='z'||m>='A'&&m<='Z')
    a++;
    else
       if(m==' ')
        b++;
      else
        if(m>='0'&&m<='9')
        c++;
        else
        d++;
 }
 printf("charaters: %d\n",a);
        printf("blanks: %d\n",b);
 printf("digitals: %d\n",c);
 printf("others: %d\n",d);
}
