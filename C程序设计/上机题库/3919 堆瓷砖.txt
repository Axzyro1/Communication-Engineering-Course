#include<stdio.h>
#define N 100
int main()
{
	int n,i,j,t,a[N],k,m;
    scanf("%d",&n);
		for(k=0;k<n;k++)
		{
		scanf("%d",&m);
		for(i=0;i<m;i++)
		{
			scanf("%d",&a[i]);
		}
		for(i=0;i<m-1;i++)
		{
              for(j=0;j<m-i-1;j++)
			{
				if (a[j]<a[j+1])
				{
				t=a[j];
				a[j]=a[j+1];
				a[j+1]=t;
				}
			}
		}
		for(i=0;i<m;i++)
		{
		printf("%d ",a[i]);
		}
		printf("\n");
	}
}
