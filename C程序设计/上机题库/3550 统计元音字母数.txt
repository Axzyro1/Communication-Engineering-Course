#include <stdio.h>
#include <string.h>
int main()
{
    char c[1000];
    int a,b,n,d,e,len,i;
    gets(c); 
    {
   	a=b=n=d=e=0;
    len = strlen(c);
    for (i=0; i<len; i++)
    {
    	if(c[i]=='a'||c[i]=='A')
    	 {a++;}
    	if(c[i]=='e'||c[i]=='E')
    	 {b++;}
    	if(c[i]=='i'||c[i]=='I')
    	 {n++;}
    	if(c[i]=='o'||c[i]=='O')
    	 {d++;}
    	if(c[i]=='u'||c[i]=='U')
         {e++;}
    }
     printf("%d\n%d\n%d\n%d\n%d\n",a,b,n,d,e);
    }
}
