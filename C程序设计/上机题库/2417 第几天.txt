#include<stdio.h>
int main()
{
    int i,j,year,month,day,MONTH[12]={31,28,31,30,31,30,31,31,30,31,30,31},D;
    while(scanf("%d-%d-%d",&year,&month,&day)!=EOF)
    {
    if((year%4==0&&year%100!=0)||year%400==0)
    {
        MONTH[1]=29;
    }
    D=0;
    for(i=0;i<month-1;i++)
    {
        D+=MONTH[i];
    }
    if(day<=MONTH[i])
    {
	D=D+day;
    printf("%d\n",D);
	}
    else
    printf("No\n");
    }
}
