#include <stdio.h>
#include <string.h>
int main()
{
int n,i,j,l,a,k;
char s[1000];
while(scanf("%d",&n)!=EOF)
{
  getchar();
  for(k=0;k<n;k++)
    {
     a=0;
     gets(s);
     l=strlen(s);
     for(i=0,j=l-1;i<l;i++,j--)
     {
       if(s[i]==s[j])
       {
         a++;
       }
     }
     if(a==l)
     printf("yes\n");
     else
     printf("no\n");
     }
 }
}
