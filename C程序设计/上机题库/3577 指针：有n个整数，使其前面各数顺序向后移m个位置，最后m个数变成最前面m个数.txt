#include <stdio.h> 
#define N 100
int move(int *a,int n,int m) 
{ 
int t;
if(*a>m)
{t=*a-m;}
else
{t=n-m+*a;} 
return t;
} 
int main() 
{ 
int i,t,m,n,a[N];  
while(scanf("%d%d",&n,&m)!=EOF) 
{
	for(i=1;i<=n;i++)
	{scanf("%d",&a[i]);}
	for(i=1;i<=n;i++)
	{
		printf("%d ",a[move(&i,n,m)]); 
	}
	printf("\n");
 } 
}
