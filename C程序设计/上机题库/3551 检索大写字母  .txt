#include <stdio.h>
#include <string.h>
int main()
{
    char str1[80],m;
    int len1,i,j;
    gets(str1); 
    {
		len1=strlen(str1);
        for (i=0;i<len1; i++)
        {
       	m=0;
        for(j=0;j<i;j++)	
    	{
		if(str1[i]==str1[j])
		m++;
        }
        if(m==0&&str1[i]>='A'&&str1[i]<='Z')
        {
       	printf("%c ",str1[i]);
        }
        }
        printf("\n");
    }
}
