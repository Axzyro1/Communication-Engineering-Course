#include <stdio.h>
#include <string.h>
int main()
{
    char c[1000],m;
    int i;
    while(gets(c)!=NULL) 
    {
	    m=0;
        for (i=0;c[i]!='\0'; i++)
        {
    	if(c[i]>c[m])
        m=i;
        }
         for (i=0;c[i]!='\0'; i++)
        {
    	if(c[i]!=c[m])
        {printf("%c",c[i]);}
        else
        {printf("%c(max)",c[i]);}
        }
        printf("\n");
    }
}
