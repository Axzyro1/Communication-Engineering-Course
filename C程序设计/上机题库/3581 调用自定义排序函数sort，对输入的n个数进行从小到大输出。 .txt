#include <stdio.h> 
#define N 100
int swap(int *a,int *b) 
{ 
int t;
t=*a;
*a=*b;
*b=t;
} 
int main() 
{ 
int i,j,n,a[N];  
while(scanf("%d",&n)!=EOF) 
{
	for(i=0;i<n;i++)
	{scanf("%d",&a[i]);}
	for(i=0;i<n-1;i++)
		{
              for(j=0;j<n-i-1;j++)
			{
				if (a[j]>a[j+1])
				{
				swap(&a[j],&a[j+1]);
				}
			}
		}
		for(i=0;i<n;i++)
		{
		printf("%d ",a[i]);
		}
		printf("\n");
	}
}
