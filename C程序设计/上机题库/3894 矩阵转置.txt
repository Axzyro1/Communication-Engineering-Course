3894 Problem C 零起点学算法93――矩阵转置 
#include<stdio.h>
#define N 100
int main()
{
  int m,n,i,j,a[N][N];
  while(scanf("%d%d",&m,&n)!=EOF)
  {
    for(i=0;i<m;i++)
     {
        for(j=0;j<n;j++)
         {
           scanf("%d",&a[i][j]);
         }
     }
      for(j=0;j<n;j++)
     {
         for(i=0;i<m;i++)
         {
           printf("%d ",a[i][j]);
         }
           printf("\n");
     }
 }
}
