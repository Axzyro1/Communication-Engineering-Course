#include <stdio.h>
#include <string.h>
int main()
{
    char str[80],m;
    int len,i,j;
    while(gets(str)!=NULL)
    {
    	m='A'+'Z';
		len=strlen(str);
        for (i=0;i<len; i++)
        {
        	if(str[i]>='A'&&str[i]<='Z')
        	printf("%c",m-str[i]);
        	else
        	printf("%c",str[i]);
        }
        printf("\n");
    }
}
