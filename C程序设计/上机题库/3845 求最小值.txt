3845 Problem D 零起点学算法44――求最小值 
#include<stdio.h>
int main()
{
	int i=0,j,n,m,min,a[i];
	while(scanf("%d",&n)!=EOF)
	{
		for(j=1;j<=n;j++)
	 {  
		while(scanf("%d",&m)!=EOF)
		{
		for(i=0;i<m;i++)
		{
			scanf("%d",&a[i]);
		}
		min=a[0];
		for(i=0;i<m;i++)
        {
		if (min>a[i])
        min=a[i];
        }
		printf("%d\n",min);
		}
 	}
	}
}
