#include<stdio.h>
#include<math.h>
int main(void)
   {

   {

   i=1;     s=0;
  while(pow(x,i)/fact(i)>=1e-5)
  {
    s+=pow(x,i)/fact(i);
     i++;
   }
   printf("%0.3lf\n",s);
   }

}
double fact(int n)
{
  double sum=1;
   while(n>0) sum*=n--;
   return sum;
}
