#include<stdio.h>
int main()
{
	int n,m,i,g,a,j;
	int prime(int x);
	scanf("%d",&n);
	for(j=0;j<n;j++)
	{scanf("%d %d",&a,&m);
		{
	if(a>=m){g=a,a=m,m=g;}
		for(i=a;i<=m;i++){
			g=prime(i);
			if(g!=0){
				printf("%d ",g);	
			}
		}
	printf("\n");
	    }
	}
}
 int prime(int x)
{
	int i,s;
	s=0;
	for(i=2;i<=(x/2);i++){
		if(x%i==0){
			s=s+i;
		}
	}
	if(s==0){return x;}
	else{return 0;}
}
