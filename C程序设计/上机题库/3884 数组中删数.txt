3884 Problem D 零起点学算法83――数组中删数 
#include<stdio.h>
int main()
{
	int i=0,j=0,n,m,a[i];
	while(scanf("%d",&n)!=EOF)
	{
		for(i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
		}
		scanf("%d",&m);
		for(i=0;i<n;i++)
		{
			if (m!=a[i])
			printf("%d ",a[i]);
			else
			break;
		}
		for(j=i+1;j<n;j++)
		{printf("%d ",a[j]);}
		printf("\n");
	}
}
