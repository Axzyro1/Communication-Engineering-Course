#include<stdio.h>
int main()
{
	int i,j=0,m;
	int prime(int x);
    for(i=100;i<200;i++)
    {
    m=prime(i);
	if(m!=0)	
	{printf("%5d",m);
	j++;
	if(j%5==0)
	{printf("\n");}
    }
    }
    printf("\n");
}

int prime(int x)
{
	int i,s;
	s=0;
	for(i=2;i<=(x/2);i++){
		if(x%i==0){
			s=s+i;
		}
	}
	if(s==0){return x;}
	else{return 0;}
}
